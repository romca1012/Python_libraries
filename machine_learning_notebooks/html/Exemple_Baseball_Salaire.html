# Dfinition de la fonction de rgle de dcision (pour l'arbre de rgrssion
def predict_value(Years, Hits):
    if Years <= 4.50:
        if Hits <= 36.50:
            if Hits <= 1.50:
                return 7.66
            else:
                return 6.23
        else:
            if Years <= 3.50:
                return 5.01
            else:
                return 5.72
    else:
        if Hits <= 117.50:
            if Years <= 6.50:
                return 5.78
            else:
                return 6.17
        else:
            if Hits <= 224.50:
                return 6.69
            else:
                return 7.59
