# Import des bibliothques
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split, KFold, GridSearchCV
from sklearn.preprocessing import OneHotEncoder, LabelEncoder
#from sklearn.tree  import DecisionTreeRegressor, plot_tree, export_text
from sklearn.metrics import r2_score
from sklearn.ensemble import RandomForestRegressor
#from sklearn.svm import SVR
import matplotlib.pyplot as plt
import seaborn as sns

data_salary =pd.read_csv('data/Salary_Data.csv')  # Lecture des donnes 
data_salary = data_salary.dropna(axis=0)  # Suppression des valeurs manquantes
cate_var_salary = data_salary.select_dtypes(['object','category'])  # Slection de variables catgorielles selon le dtype
num_var_salary = data_salary.select_dtypes(['int','float'])   # Slection de variables numriques selon le dtype
encod_cat = OneHotEncoder(sparse_output=False)  # Dfinition d'objet OnehotEncoder
var_cat_encoded_salary = encod_cat.fit_transform(cate_var_salary)   # Transformatin des variables catgorielles

df_salary_encode_cat = pd.DataFrame(
    var_cat_encoded_salary, 
    columns=encod_cat.get_feature_names_out(), 
    index=data_salary.index
)  # Conversion du data frame des variables catgoreilles encodes. Lorsqu'on fait la conversion en data frame il faut uiliser 
# les indices originaux du data frame afin d'viter des problmes dans la concatenation  cause des changements des indices dans l'Onehot encode

df_encode_num_cat = pd.concat(
    [num_var_salary, 
    df_salary_encode_cat], 
    axis=1
)

# Division des varaibles dscriptives et la variable cible
X_encod_num_cat = df_encode_num_cat.drop('Salary', axis =1)
y_encod_num_cat =  df_encode_num_cat['Salary']

X_encod_num_cat_train, X_encod_num_cat_test, y_encod_num_cat_train, y_encod_num_cat_test = train_test_split(X_encod_num_cat, y_encod_num_cat,random_state=42, test_size=0.3)

model_rf_salary =RandomForestRegressor(random_state=42)

param_search_opt_salary = {
    'n_estimators':[100, 200, 500, 700],
    'max_depth': [None, 10, 20, 30],
    #'min_samples_split': [2, 5, 10],
    #'min_samples_leaf': [1, 2, 4],
    'bootstrap': [True, False]
}
grid_search_rf_salary = GridSearchCV(
    estimator=model_rf_salary, 
    param_grid=param_search_opt_salary
)

grid_search_rf_salary.fit(X_encod_num_cat_train, y_encod_num_cat_train)

print(f"Le coefficient de dtermination (r2) optimis est : {r2_score(y_encod_num_cat_test, grid_search_rf_salary.best_estimator_.predict(X_encod_num_cat_test))}")

print(f"Les hyper-paramtres optimiss pour le modle sont:{grid_search_rf_salary.best_params_}")

# Ralisation du graphique donnes prdites vs donnes relles
plt.scatter(
    y_encod_num_cat_test,
    grid_search_rf_salary.best_estimator_.predict(X_encod_num_cat_test))

# Dfinissant les limites du graphiques pour la droite identit
x_min, x_max = plt.xlim()
y_min, y_max = plt.ylim()

# Limites des axes
axis_min = min(x_min, y_min)
axis_max = max(x_max, y_max)

# Cration du graphique indentit
identity_line = np.linspace(axis_min, axis_max)
plt.plot(identity_line, identity_line, 'k-', label='')
plt.xlabel('Valeurs relles')
plt.ylabel('Performance prdite')
plt.show()

     'importance':grid_search_rf_salary.best_estimator_.feature_importances_
    }
).sort_values('importance', ascending=False)  # Construction du data frame d'importance
